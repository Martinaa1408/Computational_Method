You are asked to implement a Python program that performs local sequence alignment between two sequences using a substitution matrix (e.g., PAM250) and a gap penalty. Your program should:
Implement a method backtracking(i, j) that recovers the best local alignment starting from a cell (i, j) with score ≥ threshold.
Implement a method find_alignments(matrix, gap, threshold) that:
-Scans the score matrix F,
-Finds all cells with value ≥ threshold,
-Performs backtracking from those cells,
-Collects and returns all alignments with their score.
In your main script, test your implementation using the PAM250 matrix, the sequences "PAWHEAE" and "HEAGAWGHEE", a gap penalty of -4, and a threshold of 5.


class SubstitutionMatrix:
    def __init__(self, filename):
        '''Read matrix from a txt file and create a dictionary'''
        with open(filename) as reader:
            characters = reader.readline().split()
            self.values = {}

            for line in reader:
                line = line.split()
                char1 = line[0]
                for char2, val in zip(characters, line[1:]):
                    self.values[char1, char2] = int(val)

    def __getitem__(self, key):
        '''Access the matrix with a key in the form (char1,char2)'''
        return self.values[key]


class SequencePair:
    def __init__(self, seq1, seq2):
        self.seq1 = seq1
        self.seq2 = seq2

    def _create_local_matrices(self, matrix, gap):
        self.F = []
        self.T = []
        for row in range(len(self.seq2) + 1):
            self.F.append([0] * (len(self.seq1) + 1))
            self.T.append([''] * (len(self.seq1) + 1))

        for row in range(1, len(self.seq2) + 1):
            for col in range(1, len(self.seq1) + 1):
                top = self.F[row - 1][col] + gap, 3, 'u'
                left = self.F[row][col - 1] + gap, 2, 'l'
                diag = self.F[row - 1][col - 1] + matrix[self.seq2[row - 1], self.seq1[col - 1]], 1, 'd'
                zero = 0, 4, ''
                self.F[row][col], _, self.T[row][col] = max(top, left, diag, zero)
